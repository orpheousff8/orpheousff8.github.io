function work(a, b) {

    console.log( a + b ); // work จะเป็น ฟังก์ชัน หรือ method ก็ได้
}
    work = spy(work);
    work(1, 2); // 3
    work(4, 5); // 9

    for (let args of work.calls) {

        alert( 'call:' + args.join() ); // "call:1,2", "call:4,5"
    }
คำตอบ 
        
function spy(func) {

    function wrapper() {

        wrapper.calls.push( Array.prototype.slice.call(arguments) );
        return func.apply(this, arguments);
    };
    
    wrapper.calls = [];
    return wrapper;
}

function f(x) {

    alert(x);
}

// ให้สร้าง decorator ฟังก์ชันที่ ครอบ f ฟังก์ชัน

let f1000 = delay(f, 1000);
let f1500 = delay(f, 1500);
f1000("test"); // แสดง "test" หลังจาก 1000ms
f1500("test"); // แสดง "test" หลังจาก 1500ms
