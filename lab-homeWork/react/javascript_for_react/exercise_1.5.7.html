  2. ถ้าเรามี Object ชื่อ salaries

    ให้สร้าง function topSalary(salaries) ที่คืนค่าชื่อคนที่มีเงินเดือนสูงสุด
    1. ถ้า salaries ไม่มีข้อมูลให้คืนค่าเป็น null
    2. ถ้าสูงสุดมีหลายคน ก็ให้คืนใครก็ได้สักคน
      3.                     
function topSalary(salaries) {

    let [...rest] = Object.entries(salaries);
    let max = rest.map((arr) => arr[1]).reduce((acc, cur) => Math.max(acc, cur), -Infinity);

    if (max === -Infinity) return null;

    return Object.keys(salaries).find((key) => salaries[key] === max);
}
