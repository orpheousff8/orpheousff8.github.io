lab ผลลัพธ์ที่ได้คืออะไร

        
function f() {
    alert(this.name);
}

f = f.bind( {name: "John"} ).bind( {name: "Ann" } );

f();
คำตอบ
John

lab ค่าของ value ในฟังก์ชันจะเปลี่ยนไปไหมหลังจาก bind

        
function sayHi() {
    alert( this.name );
}
sayHi.test = 5;

let bound = sayHi.bind({
    name: "John"
});  
alert( bound.test ); // ผลลัพธ์คืออะไร แล้ว ทำไมจึงได้อย่างนั้น
undefined เนื่องจาก bound แค่สร้าง function ใหม่จาก sayHi และมี object ใน [[BoundThis]]
แต่ไม่ได้ copy property ของ sayHi มาด้วย
